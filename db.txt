CREATE TABLE users(user_id serial PRIMARY KEY, name varchar(255) NOT NULL, email varchar(255) UNIQUE NOT NULL, password varchar(255) NOT NULL, phone varchar(20) NOT NULL, user_photo varchar(255));
CREATE TABLE recipes(recipe_id serial PRIMARY KEY, title varchar(255) NOT NULL, ingredients text, recipe_photo varchar(255), recipe_videos varchar(255), created_at timestamp, user_id int REFERENCES users(user_id));
CREATE TABLE comments( comment_id serial PRIMARY KEY, comment text, user_id int REFERENCES users(user_id), recipe_id int REFERENCES recipes(recipe_id));
